        def escape_to_latex(s)
                s2 = ""
                s.each_byte do |b|
                        if LATEX_TO_CHARCODE.include? b
                                s2 += "{\\tt \\char#{b}}" 
                        elsif LATEX_ADD_SLASH.include? b
                                s2 << ?\\ << b
                        elsif b == ?\\
                        # there is no backslash in cmr10 fonts
                                s2 += "$\\backslash$"
                        else
                                s2 << b
                        end
                end
                s2
        end
