        def maruku_error(s,src=nil,con=nil)
                policy = get_setting(:on_error)
                
                case policy
                when :ignore 
                when :raise
                        raise_error create_frame(describe_error(s,src,con))
                when :warning
                        tell_user create_frame(describe_error(s,src,con))
                else
                        raise "BugBug: policy = #{policy.inspect}"
                end
        end
