        def normalize_key_and_value(k,v)
                v = v ? v.strip : true # no value defaults to true
                k = k.strip
                
                # check synonyms
                v = true if ['yes','true'].include?(v.to_s.downcase)
                v = false if ['no','false'].include?(v.to_s.downcase)
        
                k = k.downcase.gsub(' ','_')
                return k, v
        end
