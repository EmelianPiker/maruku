        def parse_email_headers(s)
                keys={}
                match = (s =~ /((\w[\w\s]+: .*\n)+)\n/)
                if match != 0
                        keys[:data] = s
                else
                        keys[:data] = $'
                        headers = $1
                        headers.split("\n").each do |l| 
                                k, v = l.split(':')
                                k, v = normalize_key_and_value(k, v)
                                k = k.to_sym
#                               puts "K = #{k}, V=#{v}"
                                keys[k] = v
                        end
                end
                keys
        end
