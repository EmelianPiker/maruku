        def each_element(e_node_type=nil, &block) 
                @children.each do |c| 
                        if c.kind_of? MDElement
                                if (not e_node_type) || (e_node_type == c.node_type)
                                        block.call c
                                end
                                c.each_element(e_node_type, &block)
                        end
                end
        end
