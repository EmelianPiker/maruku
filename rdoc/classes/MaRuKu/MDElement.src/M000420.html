        def ==(o)
                ok = o.kind_of?(MDElement) &&
                (self.node_type == o.node_type) &&
                (self.meta_priv == o.meta_priv) &&
                (self.children == o.children)
                
                if not ok
#                       puts "This:\n"+self.inspect+"\nis different from\n"+o.inspect+"\n\n"
                end
                ok
        end
