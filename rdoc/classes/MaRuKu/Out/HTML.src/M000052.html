        def create_html_element(name, attributes_to_copy=[])
                m = Element.new name
                        (StandardAttributes+attributes_to_copy).each do |a|
                                if v = @attributes[a] then m.attributes[a.to_s] = v.to_s end
                        end
                m
        end
