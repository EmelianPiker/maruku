        def source2html(source)
                source = source.gsub(/&/,'&amp;')
                source = Text.normalize(source)
                source = source.gsub(/\&apos;/,'&#39;') # IE bug
                source = source.gsub(/'/,'&#39;') # IE bug
                Text.new(source, true, nil, true )
        end
