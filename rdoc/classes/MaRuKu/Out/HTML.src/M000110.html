        def render_png(kind, tex)
                engine = get_setting(:html_png_engine)
                method = "convert_to_png_#{engine}".to_sym
                if self.respond_to? method
                        return self.send(method, kind, tex) 
                else 
                        puts "A method called #{method} should be defined."
                        return nil
                end
        end
