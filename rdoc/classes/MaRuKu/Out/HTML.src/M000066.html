        def to_html_code_using_pre(source)
                pre = create_html_element  'pre'
                code = Element.new 'code', pre
                s = source
                
                s  = s.gsub(/&/,'&amp;')
                s = Text.normalize(s)
                s  = s.gsub(/\&apos;/,'&#39;') # IE bug
                s  = s.gsub(/'/,'&#39;') # IE bug

                if get_setting(:code_show_spaces) 
                        # 187 = raquo
                        # 160 = nbsp
                        # 172 = not
                        s.gsub!(/\t/,'&#187;'+'&#160;'*3)
                        s.gsub!(/ /,'&#172;')
                end

                text = Text.new(s, respect_ws=true, parent=nil, raw=true )
                
                code << text
                pre
        end
