        def to_html_eqref
                if eq = self.doc.eqid2eq[self.eqid]
                        num = eq.num
                        a = Element.new 'a'
                        a.attributes['class'] = 'maruku-eqref'
                        a.attributes['href'] = "#eq:#{self.eqid}"
                        a << Text.new("(#{num})")
                        a
                else
                        maruku_error "Cannot find equation #{self.eqid.inspect}"
                        Text.new "(eq:#{self.eqid})"
                end
        end
