        def add_css_to(head)
                if css_list = self.attributes[:css]
                        css_list.split.each do |css|
                        # <link type="text/css" rel="stylesheet" href="..." />
                        link = Element.new 'link'
                        link.attributes['type'] = 'text/css'
                        link.attributes['rel'] = 'stylesheet'
                        link.attributes['href'] = css
                        head << link 
                        head << xml_newline
                        end
                end
        end
