        def to_html_immediate_link
                a =  create_html_element 'a'
                url = self.url
                text = url.gsub(/^mailto:/,'') # don't show mailto
                a << Text.new(text)
                a.attributes['href'] = url
                add_class_to_link(a)
                a
        end
