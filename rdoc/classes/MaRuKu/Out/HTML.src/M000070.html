        def to_html_im_link
                if url = self.url
                        title = self.title
                        a =  wrap_as_element 'a'
                        a.attributes['href'] = url
                        a.attributes['title'] = title if title
                        return a
                else
                        maruku_error"Could not find url in #{self.inspect}"
                        tell_user "Not creating a link for ref_id = #{id.inspect}."
                        return wrap_as_element('span')
                end
        end
