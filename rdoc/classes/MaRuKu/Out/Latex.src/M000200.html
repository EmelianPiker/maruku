        def to_latex_header
                h = TexHeaders[self.level] || 'paragraph'
                
                title = children_to_latex
                if number = section_number
                        title = number + title
                end
                
                if id = self.attributes[:id]
                        # drop '#' at the beginning
                        if id[0,1] == '#' then id = [1,id.size] end
                        %{\\hypertarget{%s}{}\\%s*{{%s}}\\label{%s}\n\n} % [ id, h, title, id ]
                else
                        %{\\%s*{%s}\n\n} % [ h, title]
                end
        end
