        def to_latex_immediate_link
                url = self.url
                text = url.gsub(/^mailto:/,'') # don't show mailto
#                       gsub('~','$\sim$')
                text = latex_escape(text)
                if url[0,1] == '#'
                        url = url[1,url.size]
                        return "\\hyperlink{#{url}}{#{text}}"
                else

                        return "\\href{#{url}}{#{text}}"
                end
        end
