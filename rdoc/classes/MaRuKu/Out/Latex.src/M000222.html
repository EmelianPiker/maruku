        def to_latex_definition                
                terms = self.terms
                definitions = self.definitions
                
                s = ""
                terms.each do |t|
                        s +="\n\\item[#{t.children_to_latex}] "
                end

                definitions.each do |d|
                        s += "#{d.children_to_latex} \n"
                end
                
                s
        end
