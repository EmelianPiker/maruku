        def latex_color(s, command='color')
                if s =~ /^\#(\w\w)(\w\w)(\w\w)$/
                        r = $1.hex; g = $2.hex; b=$3.hex
                        # convert from 0-255 to 0.0-1.0
                        r = r / 255.0; g = g / 255.0; b = b / 255.0; 
                        "\\#{command}[rgb]{%0.2f,%0.2f,%0.2f}" % [r,g,b]
                elsif s =~ /^\#(\w)(\w)(\w)$/
                        r = $1.hex; g = $2.hex; b=$3.hex
                        # convert from 0-15 to 0.0-1.0
                        r = r / 15.0; g = g / 15.0; b = b / 15.0; 
                        "\\#{command}[rgb]{%0.2f,%0.2f,%0.2f}" % [r,g,b]
                else  
                        "\\#{command}{#{s}}"
                end
        end
