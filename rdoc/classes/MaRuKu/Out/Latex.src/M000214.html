        def to_latex_link
                id = self.ref_id
                ref = @doc.refs[id]
                if not ref
                        $stderr.puts "Could not find id = '#{id}'"
                        return children_to_latex
                else
                        url = ref[:url]
                        #title = ref[:title] || 'no title'

                        if url[0,1] == '#'
                                url = url[1,url.size]
                                return "\\hyperlink{#{url}}{#{children_to_latex}}"
                        else
                                return "\\href{#{url}}{#{children_to_latex}}"
                        end
                end
                
        end
