        def to_latex_inline_code; 
                source = self.raw_code
                
                # Convert to printable latex chars 
                s = latex_escape(source)
                
                color = get_setting(:code_background_color)
                colorspec = latex_color(color, 'colorbox')

                "#{colorspec}{\\tt #{s}}"
        end
