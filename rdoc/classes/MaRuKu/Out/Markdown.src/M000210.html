        def to_md_ul(context)
                len = (context[:line_length] || DefaultLineLength) - 2
                md = ""
                self.children.each_with_index do |li, i|
                        w = wrap(li.children, len-2, context)
#                       puts "W: "+ w.inspect
                        s = add_indent(w)
#                       puts "S: " +s.inspect
                        s[0,1] = "-"
                        md += s
                end
                md + "\n"
        end
