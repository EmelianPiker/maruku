        def search_abbreviations
                self.abbreviations.each do |abbrev, title|
                        reg = Regexp.new(Regexp.escape(abbrev))
                        self.replace_each_string do |s|
                                if m = reg.match(s)
                                        e = md_abbr(abbrev.dup, title ? title.dup : nil)
                                        [m.pre_match, e, m.post_match]
                                else
                                        s
                                end
                        end
                end
        end
