        def execute_code_blocks
                self.each_element(:xml_instr) do |e|
                        if e.target == 'maruku'
                                result = safe_execute_code(e, e.code)
                                if result.kind_of?(String)
                                        puts "Result is : #{result.inspect}"
                                end
                        end
                end
        end
