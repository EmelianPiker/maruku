        def read_ald(src)
                if (l=src.shift_line) =~ AttributeDefinitionList
                        id = $1;   al=$2;
                        al = read_attribute_list(CharSource.new(al,src), context=nil, break_on=[nil])
                        self.ald[id] = al;
                        return md_ald(id, al)
                else
                        maruku_error "Bug Bug:\n#{l.inspect}"
                        return nil
                end
        end
