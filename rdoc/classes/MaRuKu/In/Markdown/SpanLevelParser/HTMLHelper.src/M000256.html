                def handle_tag()
                        @already += @m.pre_match
                        @rest = @m.post_match

                        is_closing = !!@m[1]
                        tag = @m[2]
                        attributes = @m[3]

                
                        is_single = false
                        if attributes =~ /\A(.*)\/\Z/
                                attributes = $1
                                is_single = true
                        end

#                       puts "READ TAG #{@m.to_s.inspect} tag = #{tag} closing? #{is_closing} single = #{is_single}"
        
                        if TO_SANITIZE.include? tag 
                                attributes.strip!
                #             puts "Attributes: #{attributes.inspect}"
                                if attributes.size > 0
                                        @already +=  '<%s %s />' % [tag, attributes]
                                else
                                        @already +=  '<%s />' % [tag]
                                end
                        elsif is_closing
                                @already += @m.to_s
                                if @tag_stack.empty?
                                        error "Malformed: closing tag #{tag.inspect} "+
                                              "in empty list"
                                end 
                                if @tag_stack.last != tag
                                        error "Malformed: tag <#{tag}> "+
                                              "closes <#{@tag_stack.last}>"
                                end
                                @tag_stack.pop
                        else 
                                @already += @m.to_s
                                
                                @tag_stack.push(tag) unless is_single
                        end
                end
