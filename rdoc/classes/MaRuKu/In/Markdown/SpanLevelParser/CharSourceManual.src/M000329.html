                def read_regexp(r)
                        r2 = /^#{r}/
                        rest = current_remaining_buffer
                        m = r2.match(rest)
                        if m
                                @buffer_index += m.to_s.size
#                               puts "#{r} matched #{rest.inspect}: #{m.to_s.inspect}"
                        end
                        return m
                end
