        def read_text_chars(out)
                s = @buffer.size; c=nil
                while @buffer_index < s && (c=@buffer[@buffer_index]) &&
                         ((c>=?a && c<=?z) || (c>=?A && c<=?Z))
                                out << c
                                @buffer_index += 1
                end
        end
