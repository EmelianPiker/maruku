        def tell_me_the_future
                s = ""; num_e = 0;
                for i in @lines_index..@lines.size-1
                        c = case @lines[i].md_type
                                when :text; "t"
                                when :empty; num_e+=1; "e"
                                when :definition; "d"
                                else "o"
                        end
                        s += c
                        break if c == "d" or num_e>1
                end
                s     
        end
