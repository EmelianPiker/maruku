        def parse_span_better(string, parent=nil)
                if not string.kind_of? String then 
                        error "Passed #{string.class}." end

                st = (string + "")
                st.freeze
                src = CharSource.new(st, parent)
                read_span(src, EscapedCharInText, [nil])
        end
