        def read_url(src, break_on)
                if [?',?"].include? src.cur_char 
                        error 'Invalid char for url', src
                end
                
                url = read_simple(src, [], break_on)
                if not url # empty url
                        url = ""
                end
                
                if url[0] == ?< && url[-1] == ?>
                        url = url[1, url.size-2]
                end
                
                if url.size == 0 
                        return nil
                end
                
                url
        end
