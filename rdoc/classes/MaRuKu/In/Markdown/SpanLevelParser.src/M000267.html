        def read_quoted(src, con)
                case src.cur_char
                        when ?', ?"
                                quote_char = src.shift_char # opening quote
                                string = read_simple(src, EscapedCharInQuotes, [quote_char])
                                src.ignore_char # closing quote
                                return string
                        else 
#                               puts "Asked to read quote from: #{src.cur_chars(10).inspect}"
                                return nil
                end
        end
