        def extension_meta(src, con, break_on_chars)
                if m = src.read_regexp(/([^\s\:]+):/)
                        name = m[1]
                        al = read_attribute_list(src, con, break_on_chars)
#                       puts "#{name}=#{al.inspect}"
                        self.doc.ald[name] = al
                      con.push md_ald(name, al)
                else
                        al = read_attribute_list(src, con, break_on_chars)
                        self.doc.ald[name] = al
                        con.push md_ial(al)
                end
        end
