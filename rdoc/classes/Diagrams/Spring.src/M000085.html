        def Spring.parallel(s1,s2)
                pref = (s1.pref+s2.pref)*0.5
                pref = [s1.min, pref].max
                pref = [s1.max, pref].min
                pref = [s2.min, pref].max
                pref = [s2.max, pref].min
                
                Spring.new( 
                        [s1.min,s2.min].max, 
                        [s1.shrink,s2.shrink].min,
                        pref,
                        [s1.stretch,s2.stretch].min,
                        [s1.max,s2.max].min
                        )
        end
