                        def to_dot_links(links, springs)
                                s = "digraph diagram {\n"
                                links.each do |var, to_other|
                                        if not to_other.empty?
                                                var_ = var.gsub(/^[^_]+_/,'')
                                                s += "\t#{var} [label = \"#{var_}\" ];\n"
                                                to_other.each do |v2, spring|
                                                        s += "\t #{var} -> #{v2} [ label = \"#{spring.inspect}\"];\n"
                                                end
                                        end
                                end
                                s += "}\n"
                        end
