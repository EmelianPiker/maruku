                        def rec_paths(links, var1, var2)
                                all_paths = []
                                (links[var1] || []).each do |v, spring|
                                        if v == var2
                                                all_paths.push [var2]
                                        else
                                                all_paths = all_paths + rec_paths(links, v, var2)
                                        end
                                end
                                all_paths.map{|x| [var1]+x}
                        end
