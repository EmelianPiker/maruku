        def match(x,y,reg,inc_x,inc_y)
                s = ""
                best = nil
                while x<@text.width && y < @text.height
                        s << @text.get(x,y).chr
                        break if not @to_process.get(x,y)
                        if s =~ reg
                                best = s.dup 
                        end
                        x += inc_x; y += inc_y
                end
                best
        end
