        def detect_arrows
                [     [?|, /^\|*$/, [?^, ?A], [?v, ?V],  0, 1, :S],
                        [?-, /^\-*$/, [?<], [?>],  1, 0, :E],
                        [?\\,/^\\*$/, [], [], 1, 1, :SE],
                        [?/, /^\/*$/, [], [], -1, 1, :SW] 
                ].each do |char, reg, cap1, cap2, inc_x, inc_y, ty|
                        for_each_to_process(char) do |x,y,e|
                                find_arrow(x, y, reg, cap1, cap2, inc_x, inc_y)
                        end
                end
        end
