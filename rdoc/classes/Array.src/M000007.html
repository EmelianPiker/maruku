        def inspect_more(compact, join_string, add_brackets=true)
                s  = map {|x| 
                        x.kind_of?(String) ? x.inspect : 
                        x.kind_of?(MaRuKu::MDElement) ? x.inspect(compact) : 
                        (raise "WTF #{x.class} #{x.inspect}")
                }.join(join_string)
                
                add_brackets ? "[#{s}]" : s
        end
